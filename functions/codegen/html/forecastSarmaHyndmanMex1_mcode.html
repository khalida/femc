1%% SARMA forecast using method described by Hyndman in:
 2% https://www.otexts.org/fpp/8/8
 3
 4function forecast = forecastSarmaHyndmanMex(demand, ...
 5    phiValues, PhiValues, k) %#codegen
 6
 7% Assert input types
 8% NB: need lags up to p + k*P, where p is AR order, P is seasonal AR order
 9% of model and k is the seasonal period
10assert(all(size(demand) == [k+3 1]));
11
12% Pre-allocate output forecast
13forecast = zeros([k 1], 'like', demand);
14
15for thisK = 1:k;
16    if(thisK<=1)
17        forecast(thisK) =  phiValues(1)*demand(end + thisK - 1) + ...
18            phiValues(2)*demand(end + thisK - 2) + ...
19            phiValues(3)*demand(end + thisK - 3) + ...
20            PhiValues*demand(end + thisK - k) - ...
21            phiValues(1)*PhiValues*demand(end + thisK - k - 1) - ...
22            phiValues(2)*PhiValues*demand(end + thisK - k - 2) - ...
23            phiValues(3)*PhiValues*demand(end + thisK - k - 3);
24        
25    elseif(thisK<=2)
26        forecast(thisK) =  phiValues(1)*forecast(thisK - 1) + ...
27            phiValues(2)*demand(end + thisK - 2) + ...
28            phiValues(3)*demand(end + thisK - 3) + ...
29            PhiValues*demand(end + thisK - k) - ...
30            phiValues(1)*PhiValues*demand(end + thisK - k - 1) - ...
31            phiValues(2)*PhiValues*demand(end + thisK - k - 2) - ...
32            phiValues(3)*PhiValues*demand(end + thisK - k - 3);
33        
34    elseif(thisK<=3)
35        forecast(thisK) =  phiValues(1)*forecast(thisK - 1) + ...
36            phiValues(2)*forecast(thisK - 2) + ...
37            phiValues(3)*demand(end + thisK - 3) + ...
38            PhiValues*demand(end + thisK - k) - ...
39            phiValues(1)*PhiValues*demand(end + thisK - k - 1) - ...
40            phiValues(2)*PhiValues*demand(end + thisK - k - 2) - ...
41            phiValues(3)*PhiValues*demand(end + thisK - k - 3);
42        
43    elseif(thisK<=k)
44        forecast(thisK) =  phiValues(1)*forecast(thisK - 1) + ...
45            phiValues(2)*forecast(thisK - 2) + ...
46            phiValues(3)*forecast(thisK - 3) + ...
47            PhiValues*demand(end + thisK - k) - ...
48            phiValues(1)*PhiValues*demand(end + thisK - k - 1) - ...
49            phiValues(2)*PhiValues*demand(end + thisK - k - 2) - ...
50            phiValues(3)*PhiValues*demand(end + thisK - k - 3);
51        
52    elseif(thisK<=(k+1))
53        forecast(thisK) =  phiValues(1)*forecast(thisK - 1) + ...
54            phiValues(2)*forecast(thisK - 2) + ...
55            phiValues(3)*forecast(thisK - 3) + ...
56            PhiValues*forecast(thisK - k) - ...
57            phiValues(1)*PhiValues*forecast(end + thisK - k - 1) - ...
58            phiValues(2)*PhiValues*demand(end + thisK - k - 2) - ...
59            phiValues(3)*PhiValues*demand(end + thisK - k - 3);
60    elseif(thisK<=(k+2))
61        forecast(thisK) =  phiValues(1)*forecast(thisK - 1) + ...
62            phiValues(2)*forecast(thisK - 2) + ...
63            phiValues(3)*forecast(thisK - 3) + ...
64            PhiValues*forecast(thisK - k) - ...
65            phiValues(1)*PhiValues*forecast(end + thisK - k - 1) - ...
66            phiValues(2)*PhiValues*forecast(end + thisK - k - 2) - ...
67            phiValues(3)*PhiValues*demand(end + thisK - k - 3);
68    else
69        forecast(thisK) =  phiValues(1)*forecast(thisK - 1) + ...
70            phiValues(2)*forecast(thisK - 2) + ...
71            phiValues(3)*forecast(thisK - 3) + ...
72            PhiValues*forecast(thisK - k) - ...
73            phiValues(1)*PhiValues*forecast(end + thisK - k - 1) - ...
74            phiValues(2)*PhiValues*forecast(end + thisK - k - 2) - ...
75            phiValues(3)*PhiValues*forecast(end + thisK - k - 3);
76    end
77end
78end
